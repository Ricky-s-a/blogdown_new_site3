# データの準備

df <- data.frame(x =  1:3, y = 3:1, z = c(letters[1:3]))
area <- ggplot(df, aes(x, y)) +
  geom_bar(aes(fill = z), stat = "identity") +
  theme(legend.position = "none") +
  labs(x = NULL, y = NULL)
Colors <- c("Set3", "Set2", "Set1")

# 複数の色合い同時に表示させる

Areas <- 
  purrr::map(.x = Colors, .f = ~{area + scale_fill_brewer(palette = .x)}) %>% 
  purrr::reduce(`+`)
Areas
