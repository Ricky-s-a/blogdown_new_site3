関数を作成

kakezan2 <- function(n) {
  result2 <- matrix(nrow = n, ncol = n)
  for (i in 1:n) {
    result2[i,] <- seq(i, by = i, length.out = n)
  }
  colnames(result2) <- c(1:n)
  rownames(result2) <- c(1:n)
  print(result2)
}
